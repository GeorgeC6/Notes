### xçš„çº¿æ€§å˜æ¢
### w[0][0]+w[1][0]*x = z1
### w[0][1]+w[1][1]*x = z2

### w[0][3]+w[1][3]*x = z4

trans=x_c@w

### sigmoid(zi)
ns=4
for i in range(ns):
    plt.scatter(x,expit(trans[:,i]))

### z=sigmoid(z1)+sigmoid(z2)+sigmoid(z3)

out=(middle-np.mean(middle))/np.std(middle)

### y = sigmoid(z)
plt.plot(x,expit(out))
plt.plot(x,expit(out)>0.5)
